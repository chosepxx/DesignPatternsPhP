create table cliente (
    id_cliente int PRIMARY key AUTO_INCREMENT,
    nombre varchar (100),
    apellidos varchar (100), 
    numero_celular varchar (10),
    correo varchar(100),
    fecha_nacimiento date,
    sexo varchar (2),
    nacionalidad varchar(60),
    responsabilidad_social varchar(100)
);


create table empleado (
    id_empleado int PRIMARY key AUTO_INCREMENT,
    nombre varchar (100),
    apellidos varchar (100), 
    numero_celular varchar (10),
    correo varchar(100),
    fecha_nacimiento date,
    puesto varchar (60),
    sexo varchar (2),
    nacionalidad varchar(60),
    responsabilidad_social varchar(100),
    lugar_residencia varchar(200),
    estado_civil varchar (40)
);

create table productos (
    id_producto int PRIMARY key AUTO_INCREMENT,
    base_pintura varchar(40),
    precio int, 
    area_aplicacion varchar(50),
    fecha_caducidad date, 
    color_pintura varchar (40),
    cantidad_stock int,
    max int, 
    min int,
    descripcion varchar (300), 
    marca varchar (40)
);


create table registro_pintura(
    id_registro int PRIMARY KEY AUTO_INCREMENT,
    fecha_compra date,
    id_cliente int,
    id_empleado int,
    id_producto int,
    base varchar (40),
    acabado varchar (40),
    formula_color varchar (60),
    tama単o_envase varchar (44),
    FOREIGN KEY(id_cliente) REFERENCES cliente(id_cliente),
    FOREIGN KEY(id_empleado) REFERENCES empleado(id_empleado),
    FOREIGN KEY(id_producto) REFERENCES empleado(id_producto)

);
